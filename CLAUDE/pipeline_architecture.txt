SHARED STATE DICT (passed through all agents):
{
  "input":          str,   # original user query
  "plan":           list,  # planner phases
  "literature":     dict,  # papers, abstracts, key findings
  "patents":        dict,  # patent claims, scaffolds
  "target_info":    dict,  # UniProt, PDB, gene name
  "hypothesis":     dict,  # selected target + MoA statement
  "candidates":     list,  # generated SMILES + properties
  "docking_results":list,  # binding scores per candidate
  "synthesis":      dict,  # SA scores + retrosynthetic routes
  "report":         str,   # final compiled Markdown
  "errors":         list,  # captured non-fatal errors
  "metadata":       dict,  # timestamps, run ID
}
